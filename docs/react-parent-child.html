👨‍👧‍👦 Parent-Child Communication

Master props, callbacks, and data flow between React components. Essential for building complex component trees.

React Data Flow Principles

React follows a one-way data flow:

┌─────────────────────────────────────────┐
│ PARENT COMPONENT                        │
│ ┌────────────────────────────────────┐  │
│ │ State: products = [...]            │  │
│ │                                    │  │
│ │ ┌──────────────────────────────┐   │  │
│ │ │ CHILD 1                      │   │  │
│ │ │ Receives: products (props)   │   │  │
│ │ │ Sends: onSelect callback     │   │  │
│ │ └──────────────────────────────┘   │  │
│ │                                    │  │
│ │ ┌──────────────────────────────┐   │  │
│ │ │ CHILD 2                      │   │  │
│ │ │ Receives: selected (props)   │   │  │
│ │ │ Displays selected product    │   │  │
│ │ └──────────────────────────────┘   │  │
│ └────────────────────────────────────┘  │
└─────────────────────────────────────────┘

DATA FLOW:
Parent → Children (props)
Children → Parent (callback functions)

Method 1: Props (Parent to Child)

The primary way to pass data from parent to child. Props are read-only and flow downward.

Simple Props Example

// Parent Component
function Parent() {
  const [selectedProduct, setSelectedProduct] = useState(null);

  return (
    <div>
      <h1>Product Store</h1>

      <!-- Pass data as props -->
      <ProductCard
        name="Laptop"
        price={999.99}
        inStock={true}
      />
    </div>
  );
}

// Child Component
function ProductCard(props) {
  // Access props
  return (
    <div className="card">
      <h2>{props.name}</h2>
      <p>${props.price}</p>
      <p>{props.inStock ? 'In Stock' : 'Out of Stock'}</p>
    </div>
  );
}

// Or use destructuring (cleaner)
function ProductCard({ name, price, inStock }) {
  return (
    <div className="card">
      <h2>{name}</h2>
      <p>${price}</p>
      <p>{inStock ? 'In Stock' : 'Out of Stock'}</p>
    </div>
  );
}

Method 2: Callback Functions (Child to Parent)

When a child needs to send data back to parent, the parent passes a callback function as a prop.

Real Example: ProductCatalog with Filtering

// Parent Component - Has state and passes callbacks to children
function ProductCatalog() {
  const [products, setProducts] = useState([
    { id: 1, name: 'Laptop', category: 'Electronics', price: 999 },
    { id: 2, name: 'Mouse', category: 'Electronics', price: 29 },
    { id: 3, name: 'Desk', category: 'Furniture', price: 299 }
  ]);

  const [selectedCategory, setSelectedCategory] = useState('All');

  // Filter logic
  const filteredProducts = selectedCategory === 'All'
    ? products
    : products.filter(p => p.category === selectedCategory);

  // Callback function passed to child
  const handleCategoryChange = (category) => {
    setSelectedCategory(category);
  };

  return (
    <div className="catalog">
      <h1>Product Catalog</h1>

      <!-- Pass callback to FilterPanel -->
      <FilterPanel
        categories={['Electronics', 'Furniture']}
        selectedCategory={selectedCategory}
        onCategoryChange={handleCategoryChange}
      />

      <!-- Pass filtered data to ProductList -->
      <ProductList products={filteredProducts} />
    </div>
  );
}

// Child 1: Filter panel - calls parent's callback
function FilterPanel({
  categories,
  selectedCategory,
  onCategoryChange
}) {
  return (
    <div className="filter-panel">
      <h3>Filter by Category</h3>
      <button
        onClick={() => onCategoryChange('All')}
        className={selectedCategory === 'All' ? 'active' : ''}
      >
        All
      </button>

      {categories.map(cat => (
        <button
          key={cat}
          onClick={() => onCategoryChange(cat)}
          className={selectedCategory === cat ? 'active' : ''}
        >
          {cat}
        </button>
      ))}
    </div>
  );
}

// Child 2: Product list - displays filtered data
function ProductList({ products }) {
  return (
    <div className="product-list">
      {products.map(product => (
        <div key={product.id} className="product">
          <h4>{product.name}</h4>
          <p>${product.price}</p>
          <p>Category: {product.category}</p>
        </div>
      ))}
    </div>
  );
}

Method 3: Lifting State Up

When multiple children need to share state, move the state to their common parent.

Example: Product Details with Add to Cart

// Parent - holds shared state
function App() {
  const [cart, setCart] = useState([]);
  const [product] = useState({
    id: 1,
    name: 'Wireless Mouse',
    price: 29.99,
    description: 'High precision wireless mouse'
  });

  // Callback for when child adds to cart
  const handleAddToCart = (productToAdd) => {
    setCart([...cart, productToAdd]);
  };

  return (
    <div className="app">
      <!-- Pass product and callback -->
      <ProductDetail
        product={product}
        onAddToCart={handleAddToCart}
      />

      <!-- Pass cart data -->
      <ShoppingCart items={cart} />
    </div>
  );
}

// Child 1: Shows product and button
function ProductDetail({ product, onAddToCart }) {
  return (
    <div className="product-detail">
      <h2>{product.name}</h2>
      <p>{product.description}</p>
      <p className="price">${product.price}</p>

      <!-- Call parent's callback when clicked -->
      <button onClick={() => onAddToCart(product)}>
        Add to Cart
      </button>
    </div>
  );
}

// Child 2: Shows cart
function ShoppingCart({ items }) {
  const total = items.reduce((sum, item) => sum + item.price, 0);

  return (
    <div className="cart">
      <h3>Cart ({items.length} items)</h3>
      {items.map((item, idx) => (
        <div key={idx}>{item.name} - ${item.price}</div>
      ))}
      <h4>Total: ${total.toFixed(2)}</h4>
    </div>
  );
}

The Prop Drilling Problem

When data needs to flow through many levels of components just to reach a deeply nested child.

// ❌ Bad: Prop drilling through 5 levels
function App() {
  const [user] = useState({ name: 'John', role: 'admin' });
  return <Level1 user={user} />;
}

function Level1({ user }) {
  return <Level2 user={user} />;
}

function Level2({ user }) {
  return <Level3 user={user} />;
}

function Level3({ user }) {
  return <Level4 user={user} />;
}

function Level4({ user }) {
  return <Level5 user={user} />;
}

// Finally need the data!
function Level5({ user }) {
  return <div>Welcome {user.name}</div>;
}

Solutions to Prop Drilling

Solution 1: Context API (Built-in)

// Create context
const UserContext = React.createContext();

// Wrap app with provider
function App() {
  const [user] = useState({ name: 'John', role: 'admin' });

  return (
    <UserContext.Provider value={user}>
      <Level1 />
    </UserContext.Provider>
  );
}

// Skip to the component that needs it - no drilling!
function Level5() {
  const user = useContext(UserContext);
  return <div>Welcome {user.name}</div>;
}

Solution 2: Redux (For Complex State)

Use Redux for global state management (see redux.html)

Solution 3: Composition (Pass JSX as Props)

// Instead of passing data through levels,
// pass complete components as children

function App() {
  const [user] = useState({ name: 'John' });

  return (
    <Layout>
      <Header user={user} />
      <Content />
      <Footer />
    </Layout>
  );
}

// Layout doesn't need to know about user
function Layout({ children }) {
  return (
    <div className="layout">
      {children}
    </div>
  );
}

Best Practices

  1. Keep Props Simple: Don't pass unnecessary props 
    // ❌ Bad - passing whole object
<ProductCard product={product} />
// Then accessing: product.name, product.price, product.id

// ✅ Better - pass only what you need
<ProductCard name={name} price={price} />
  2. Use Destructuring: Cleaner and documents expectations 
    // ❌ Bad
function Card(props) { return <h1>{props.title}</h1>; }

// ✅ Good
function Card({ title, description, price }) {
  return <div>...</div>;
}
  3. Name Callbacks Clearly: Use on[Event] pattern 
    // ❌ Unclear
<Button onClick={handleClick} />

// ✅ Clear
<Button onAddToCart={handleAddToCart} />
<Button onDelete={handleDelete} />
  4. Consider Performance: Avoid passing new functions on every render 
    // ❌ Bad - new function every render
<Child onClick={() => handleClick(id)} />

// ✅ Better - use useCallback
const memoizedCallback = useCallback(() => {
  handleClick(id);
}, [id]);
<Child onClick={memoizedCallback} />

Interview Q&A

Q1: How do you pass data from child to parent in React?

Answer: Parent passes a callback function as a prop. When child needs to send data, it calls the callback with the data as an argument.

// Parent
function Parent() {
  const handleChildData = (data) => console.log(data);
  return <Child onSendData={handleChildData} />;
}

// Child calls the callback
function Child({ onSendData }) {
  return <button onClick={() => onSendData('Hello Parent')}>
    Send Data
  </button>;
}

Q2: What is prop drilling and why is it a problem?

Answer: Prop drilling is passing data through multiple intermediate components just to reach a nested child that needs it. Problems:

  • Hard to maintain - don't know which props are actually used where
  • Performance - unnecessary re-renders of intermediate components
  • Fragile - renaming props requires changes through entire tree

Solutions: Context API, Redux, component composition

Q3: When should you lift state up?

Answer: Lift state up to the lowest common ancestor when:

  • Multiple children need access to same state
  • One child's state affects another child
  • Parent needs to coordinate between children

Example: If ProductDetail and ShoppingCart both need to know what's selected, lift selection state to App (their common parent).
