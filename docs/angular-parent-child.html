👨‍👧‍👦 Parent-Child Communication

Master @Input, @Output, EventEmitter, and template references for component communication in Angular.

Angular Component Communication Methods

Method Direction Use Case
@Input Parent → Child Pass data from parent to child
@Output + EventEmitter Child → Parent Send events/data from child to parent
@ViewChild Parent accesses Child Direct access to child component/element
Services + Subjects Any → Any Complex communication, siblings

Method 1: @Input (Parent to Child)

Pass data from parent to child component using @Input decorator.

Simple @Input Example

// Parent Component
import { Component } from '@angular/core';

@Component({
  selector: 'app-parent',
  template: `
    <div>
      <h1>Parent Component</h1>

      <!-- Pass data as @Input -->
      <app-product-card
        [productName]="'Laptop'"
        [productPrice]="999.99"
        [inStock]="true"
      ></app-product-card>
    </div>
  `
})
export class ParentComponent {}

// Child Component
import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-product-card',
  template: `
    <div class="card">
      <h2>{{ productName }}</h2>
      <p>Price: ${{ productPrice }}</p>
      <p>{{ inStock ? 'In Stock' : 'Out of Stock' }}</p>
    </div>
  `
})
export class ProductCardComponent {
  @Input() productName: string = '';
  @Input() productPrice: number = 0;
  @Input() inStock: boolean = false;
}

Method 2: @Output + EventEmitter (Child to Parent)

Send events from child to parent using @Output and EventEmitter.

Real Example: ProductCatalog with Category Filter

// Parent Component
import { Component } from '@angular/core';

@Component({
  selector: 'app-catalog',
  template: `
    <div class="catalog">
      <h1>Product Catalog</h1>

      <!-- Listen for onCategoryChange event from child -->
      <app-filter-panel
        (onCategoryChange)="handleCategoryChange($event)"
      ></app-filter-panel>

      <!-- Display filtered products -->
      <app-product-list
        [products]="filteredProducts"
        (onAddToCart)="addToCart($event)"
      ></app-product-list>
    </div>
  `
})
export class CatalogComponent {
  products = [
    { id: 1, name: 'Laptop', category: 'Electronics', price: 999 },
    { id: 2, name: 'Mouse', category: 'Electronics', price: 29 },
    { id: 3, name: 'Desk', category: 'Furniture', price: 299 }
  ];

  selectedCategory = 'All';
  filteredProducts = this.products;

  handleCategoryChange(category: string) {
    this.selectedCategory = category;
    // Filter products
    this.filteredProducts = category === 'All'
      ? this.products
      : this.products.filter(p => p.category === category);
  }

  addToCart(product: any) {
    console.log('Added to cart:', product);
  }
}

// Child 1: Filter Panel - emits events
import { Component, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-filter-panel',
  template: `
    <div class="filter-panel">
      <h3>Filter by Category</h3>
      <button
        *ngFor="let cat of categories"
        (click)="selectCategory(cat)"
        [class.active]="selectedCategory === cat"
      >
        {{ cat }}
      </button>
    </div>
  `
})
export class FilterPanelComponent {
  @Output() onCategoryChange = new EventEmitter<string>();

  categories = ['All', 'Electronics', 'Furniture'];
  selectedCategory = 'All';

  selectCategory(category: string) {
    this.selectedCategory = category;
    // Emit event to parent
    this.onCategoryChange.emit(category);
  }
}

// Child 2: Product List
import { Component, Input, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-product-list',
  template: `
    <div class="product-list">
      <div *ngFor="let product of products" class="product">
        <h4>{{ product.name }}</h4>
        <p>${{ product.price }}</p>
        <p>Category: {{ product.category }}</p>
        <button (click)="addToCart(product)">
          Add to Cart
        </button>
      </div>
    </div>
  `
})
export class ProductListComponent {
  @Input() products: any[] = [];
  @Output() onAddToCart = new EventEmitter<any>();

  addToCart(product: any) {
    this.onAddToCart.emit(product);
  }
}

Method 3: @ViewChild (Parent Accesses Child)

Directly access child component methods and properties from parent.

Example: Timer Control

// Parent Component
import { Component, ViewChild } from '@angular/core';
import { TimerComponent } from './timer.component';

@Component({
  selector: 'app-parent',
  template: `
    <div>
      <h1>Timer Controls</h1>

      <!-- Template reference variable #timer -->
      <app-timer #timer></app-timer>

      <!-- Call child methods directly -->
      <button (click)="startTimer()">Start</button>
      <button (click)="stopTimer()">Stop</button>
      <button (click)="resetTimer()">Reset</button>
      <button (click)="showElapsedTime()">Show Time</button>
    </div>
  `
})
export class ParentComponent {
  // @ViewChild gets reference to child component
  @ViewChild('timer') timerComponent!: TimerComponent;

  startTimer() {
    this.timerComponent.start();
  }

  stopTimer() {
    this.timerComponent.stop();
  }

  resetTimer() {
    this.timerComponent.reset();
  }

  showElapsedTime() {
    const elapsed = this.timerComponent.getElapsedTime();
    alert(`Elapsed: ${elapsed}s`);
  }
}

// Child Component: Timer
import { Component } from '@angular/core';

@Component({
  selector: 'app-timer',
  template: `
    <div class="timer">
      <h2>Timer: {{ seconds }}s</h2>
    </div>
  `
})
export class TimerComponent {
  seconds = 0;
  private interval: any;

  start() {
    if (!this.interval) {
      this.interval = setInterval(() => {
        this.seconds++;
      }, 1000);
    }
  }

  stop() {
    if (this.interval) {
      clearInterval(this.interval);
      this.interval = null;
    }
  }

  reset() {
    this.seconds = 0;
    this.stop();
  }

  getElapsedTime(): number {
    return this.seconds;
  }
}

Method 4: Two-Way Binding with [(ngModel)]

Combine @Input and @Output for two-way data binding.

// Parent Component
@Component({
  template: `
    <app-counter [(count)]="myCount"></app-counter>
    <p>Count from parent: {{ myCount }}</p>
  `
})
export class ParentComponent {
  myCount = 0;
}

// Child Component with two-way binding
@Component({
  selector: 'app-counter',
  template: `
    <div>
      <button (click)="increment()">+</button>
      <span>{{ count }}</span>
      <button (click)="decrement()">-</button>
    </div>
  `
})
export class CounterComponent {
  @Input() count: number = 0;
  @Output() countChange = new EventEmitter<number>();

  increment() {
    this.count++;
    this.countChange.emit(this.count);
  }

  decrement() {
    this.count--;
    this.countChange.emit(this.count);
  }
}

Angular vs React: Parent-Child Communication

Scenario Angular React
Parent → Child @Input decorator Props
Child → Parent @Output + EventEmitter Callback function props
Parent accesses child @ViewChild useRef hook
Two-way binding [(ngModel)] or Custom useState + onChange

Best Practices

  1. Use @Input for data, @Output for events: Keep the component contract clear 
    // ✅ Good - clear separation
@Input() productName: string = '';
@Output() onDelete = new EventEmitter<string>();

// ❌ Wrong - using @Output for data flow
@Output() productData = new EventEmitter<Product>();
  2. Avoid @ViewChild when possible: Use @Output/EventEmitter instead 
    // ❌ Bad - tight coupling
@ViewChild(ChildComponent) child!: ChildComponent;
this.child.doSomething();

// ✅ Better - loose coupling via @Output
@Output() onEvent = new EventEmitter();
  3. Type your @Output EventEmitter: Provides type safety 
    // ❌ Bad - no type
@Output() onSelect = new EventEmitter();

// ✅ Good - typed
@Output() onSelect = new EventEmitter<Product>();
  4. Consider OnChanges for @Input updates: Detect when inputs change 
    import { OnChanges, SimpleChanges } from '@angular/core';

export class ChildComponent implements OnChanges {
  @Input() productId: number = 0;

  ngOnChanges(changes: SimpleChanges) {
    if (changes['productId']) {
      // React to productId change
      this.loadProduct(this.productId);
    }
  }
}

Interview Q&A

Q1: What's the difference between @Input and @Output?

Answer:

  • @Input: Receives data FROM parent component. Used with property binding [property]="value"
  • @Output: Sends events/data TO parent component. Used with event binding (event)="handler($event)"
// Child receives data via @Input
@Input() productName: string = '';

// Child sends data via @Output
@Output() onSelect = new EventEmitter<Product>();

Q2: How do you pass data from child to parent in Angular?

Answer: Using @Output decorator with EventEmitter. Child emits an event with data, parent listens to it.

// Child
@Output() onSelect = new EventEmitter<Product>();
selectProduct(product: Product) {
  this.onSelect.emit(product);
}

// Parent listens
<app-child (onSelect)="handleSelect($event)"></app-child>
handleSelect(product: Product) {
  console.log(product);
}

Q3: When should you use @ViewChild vs @Output?

Answer:

  • Use @Output: For event-driven communication, loose coupling, standard component communication
  • Use @ViewChild: When you need direct access to child methods/properties, for imperative operations
  • Prefer @Output: It's the Angular way and maintains clean component contracts
