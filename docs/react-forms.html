📋 Forms & Validation

Handle forms, input validation, and form submission in React.

Working with Forms in React

React forms are controlled components where React manages the form state. Unlike HTML forms, React inputs don't automatically handle state - you control them via state variables and onChange handlers.

This gives you full control over form validation, submission, and user experience.

Why Forms Matter

  • Real-time Validation: Validate as user types
  • State Control: Know exactly what user entered
  • Better UX: Clear error messages, disabled submit, etc.
  • Flexibility: Custom validation rules for your app
  • Security: Validate before sending to server

Key Concepts

🎮 Controlled Components

React state controls the input value:

const [email, setEmail] = useState('');

<input
  value={email}
  onChange={e => setEmail(e.target.value)}
/>

✅ Form Validation

  • Required fields check
  • Email format validation
  • Password strength rules
  • Minimum length checks

📤 Form Submission

Handle submission with preventDefault:

const handleSubmit = (e) => {
  e.preventDefault();
  // Send data to API
};

Code Examples

Example 1: Simple Login Form

import { useState } from 'react';

function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');

  const handleSubmit = async (e) => {
    e.preventDefault();

    // Validate
    if (!email || !password) {
      setError('All fields required');
      return;
    }

    // Send to API
    try {
      const response = await fetch('http://localhost:5000/api/login', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ email, password })
      });

      if (response.ok) {
        console.log('Login successful!');
      } else {
        setError('Invalid credentials');
      }
    } catch (err) {
      setError('Error: ' + err.message);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>Email</label>
        <input
          type="email"
          value={email}
          onChange={e => setEmail(e.target.value)}
        />
      </div>

      <div>
        <label>Password</label>
        <input
          type="password"
          value={password}
          onChange={e => setPassword(e.target.value)}
        />
      </div>

      {error && <p className="error">{error}</p>}

      <button type="submit">Login</button>
    </form>
  );
}

Example 2: Form with Validation

function ProductForm() {
  const [form, setForm] = useState({
    name: '',
    price: '',
    category: ''
  });
  const [errors, setErrors] = useState({});

  const validateForm = () => {
    const newErrors = {};

    if (!form.name.trim()) {
      newErrors.name = 'Name is required';
    }

    if (!form.price || parseFloat(form.price) <= 0) {
      newErrors.price = 'Price must be positive';
    }

    if (!form.category) {
      newErrors.category = 'Category is required';
    }

    return newErrors;
  };

  const handleChange = (e) => {
    const { name, value } = e.target;
    setForm(prev => ({ ...prev, [name]: value }));
    // Clear error for this field
    setErrors(prev => ({ ...prev, [name]: '' }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();

    const newErrors = validateForm();
    if (Object.keys(newErrors).length > 0) {
      setErrors(newErrors);
      return;
    }

    console.log('Form valid, submitting:', form);
    // Send to API
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>Product Name</label>
        <input
          name="name"
          value={form.name}
          onChange={handleChange}
        />
        {errors.name && <span className="error">{errors.name}</span>}
      </div>

      <div>
        <label>Price</label>
        <input
          type="number"
          name="price"
          value={form.price}
          onChange={handleChange}
        />
        {errors.price && <span className="error">{errors.price}</span>}
      </div>

      <div>
        <label>Category</label>
        <select name="category" value={form.category} onChange={handleChange}>
          <option value="">Select...</option>
          <option value="electronics">Electronics</option>
          <option value="clothing">Clothing</option>
        </select>
        {errors.category && <span className="error">{errors.category}</span>}
      </div>

      <button type="submit">Add Product</button>
    </form>
  );
}

Example 3: Password Strength Indicator

function SignupForm() {
  const [password, setPassword] = useState('');
  const [showPassword, setShowPassword] = useState(false);

  const getPasswordStrength = (pwd) => {
    if (!pwd) return 'weak';

    let strength = 'weak';
    if (pwd.length >= 8 && /[0-9]/.test(pwd) && /[A-Z]/.test(pwd)) {
      strength = 'medium';
    }
    if (pwd.length >= 12 && /[!@#$%^&*]/.test(pwd)) {
      strength = 'strong';
    }

    return strength;
  };

  const strength = getPasswordStrength(password);

  return (
    <div>
      <input
        type={showPassword ? 'text' : 'password'}
        value={password}
        onChange={e => setPassword(e.target.value)}
        placeholder="Enter password"
      />

      <button
        type="button"
        onClick={() => setShowPassword(!showPassword)}
      >
        {showPassword ? 'Hide' : 'Show'}
      </button>

      <!-- Strength indicator -->
      <div>
        <div className={`strength-bar ${strength}`}></div>
        <p>Strength: {strength}</p>
      </div>

      <!-- Requirements checklist -->
      <ul>
        <li className={password.length >= 8 ? 'ok' : 'bad'}>
          At least 8 characters
        </li>
        <li className={/[0-9]/.test(password) ? 'ok' : 'bad'}>
          Has number
        </li>
        <li className={/[A-Z]/.test(password) ? 'ok' : 'bad'}>
          Has uppercase
        </li>
      </ul>
    </div>
  );
}

✅ Best Practices

1. Use Controlled Components

Why: React controls the state, not the browser

2. Validate on Submit AND Change

Why: Catch errors early, but don't overwhelm user

3. Use preventDefault()

Why: Stop browser from submitting form normally

4. Show Clear Error Messages

Why: Helps users fix issues quickly

🎓 Interview Q&A

Q1: What's the difference between controlled and uncontrolled components?

Answer: Controlled components are managed by React state. Uncontrolled let the browser manage state. Use controlled for validation and control.

Q2: When should you validate forms?

Answer: On change (show feedback), on blur (not too intrusive), on submit (final check)

Q3: How do you handle complex forms with many fields?

Answer: Use useReducer for state or form libraries like React Hook Form/Formik
