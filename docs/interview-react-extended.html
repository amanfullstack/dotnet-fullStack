Q1: What is React and JSX?

Click to expand answer

Answer: React is a JavaScript library for building UIs with reusable components. JSX is syntax that looks like HTML but compiles to JavaScript.

// JSX - looks like HTML
const App = () => {
  return (
    <div className="container">
      <h1>Hello, React!</h1>
      <p>Build UIs declaratively</p>
    </div>
  );
};

// Compiles to:
const App = () => {
  return React.createElement(
    'div',
    { className: 'container' },
    React.createElement('h1', null, 'Hello, React!'),
    React.createElement('p', null, 'Build UIs declaratively')
  );
};

Q3: What is useState Hook and how to use it?

Click to expand answer

Answer: useState manages component state (reactive data). Returns [state, setState] tuple.

import { useState } from 'react';

const ProductForm = () => {
  // Simple state
  const [name, setName] = useState('');

  // Complex state
  const [product, setProduct] = useState({
    name: '',
    price: 0,
    category: ''
  });

  // Counter
  const [count, setCount] = useState(0);

  const handleNameChange = (e) => {
    setName(e.target.value);
  };

  const handleProductChange = (field, value) => {
    setProduct(prev => ({
      ...prev,
      [field]: value
    }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    console.log(product);
    // Reset form
    setProduct({ name: '', price: 0, category: '' });
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        value={product.name}
        onChange={(e) => handleProductChange('name', e.target.value)}
        placeholder="Product name"
      />
      <input
        type="number"
        value={product.price}
        onChange={(e) => handleProductChange('price', parseFloat(e.target.value))}
        placeholder="Price"
      />
      <button type="submit">Submit</button>
    </form>
  );
};

Key Rules: State updates queue, functional updates prevent stale closures

Q4: What is useEffect Hook and cleanup?

Click to expand answer

Answer: useEffect runs side effects (API calls, subscriptions) after render. Dependencies control when it runs.

import { useEffect, useState } from 'react';

const ProductList = () => {
  const [products, setProducts] = useState([]);
  const [loading, setLoading] = useState(true);

  // Run once on mount (like componentDidMount)
  useEffect(() => {
    fetchProducts();
  }, []); // Empty dependency array

  // Run when dependency changes
  useEffect(() => {
    console.log('Products changed:', products);
  }, [products]); // Dependency array

  // Cleanup function (like componentWillUnmount)
  useEffect(() => {
    const timer = setInterval(() => {
      console.log('Checking for updates...');
    }, 5000);

    // Cleanup
    return () => {
      clearInterval(timer);
    };
  }, []);

  const fetchProducts = async () => {
    try {
      setLoading(true);
      const response = await fetch('http://localhost:5000/api/products');
      const data = await response.json();
      setProducts(data);
    } catch (err) {
      console.error('Error fetching products:', err);
    } finally {
      setLoading(false);
    }
  };

  if (loading) return <p>Loading...</p>;

  return (
    <div>
      {products.map(p => <ProductCard key={p.id} product={p} />)}
    </div>
  );
};

Dependency Array: [] = run once, [dep] = run when dep changes, no array = run every render

Q5: What is useContext and Context API?

Click to expand answer

Answer: Context API shares data between components without prop drilling. useContext reads the context.

import { createContext, useContext, useState } from 'react';

// 1. Create context
const ThemeContext = createContext();

// 2. Create provider component
export const ThemeProvider = ({ children }) => {
  const [theme, setTheme] = useState('light');

  const toggleTheme = () => {
    setTheme(prev => prev === 'light' ? 'dark' : 'light');
  };

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
};

// 3. Create custom hook to use context
export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
};

// 4. Use in component (no prop drilling!)
const Header = () => {
  const { theme, toggleTheme } = useTheme();

  return (
    <header style={{ background: theme === 'light' ? 'white' : 'black' }}>
      <button onClick={toggleTheme}>
        {theme === 'light' ? '🌙' : '☀️'}
      </button>
    </header>
  );
};

// 5. Wrap app with provider (App.jsx)
<ThemeProvider>
  <Header />
</ThemeProvider>

Q6: What are Custom Hooks?

Click to expand answer

Answer: Custom hooks are reusable functions with hooks logic. Extract logic from components.

// hooks/useProduct.js
import { useState, useEffect } from 'react';

export const useProduct = () => {
  const [products, setProducts] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const API_URL = 'http://localhost:5000/api/products';

  // Fetch all products
  const fetchProducts = async () => {
    setLoading(true);
    try {
      const response = await fetch(API_URL);
      const data = await response.json();
      setProducts(data);
      setError(null);
    } catch (err) {
      setError('Failed to fetch products');
    } finally {
      setLoading(false);
    }
  };

  // Create product
  const createProduct = async (product) => {
    try {
      const response = await fetch(API_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(product)
      });
      const data = await response.json();
      setProducts([...products, data]);
      return data;
    } catch (err) {
      setError('Failed to create product');
    }
  };

  // Update product
  const updateProduct = async (id, product) => {
    try {
      const response = await fetch(`${API_URL}/${id}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(product)
      });
      const data = await response.json();
      setProducts(products.map(p => p.id === id ? data : p));
      return data;
    } catch (err) {
      setError('Failed to update product');
    }
  };

  // Delete product
  const deleteProduct = async (id) => {
    try {
      await fetch(`${API_URL}/${id}`, { method: 'DELETE' });
      setProducts(products.filter(p => p.id !== id));
    } catch (err) {
      setError('Failed to delete product');
    }
  };

  useEffect(() => {
    fetchProducts();
  }, []);

  return {
    products,
    loading,
    error,
    fetchProducts,
    createProduct,
    updateProduct,
    deleteProduct
  };
};

// Usage in component
const ProductList = () => {
  const { products, loading, createProduct } = useProduct();

  return (
    <div>
      {loading ? <p>Loading...</p> : products.map(p => ...)}
    </div>
  );
};

Q7: How do you handle forms and validation?

Click to expand answer 
const ProductForm = ({ onSubmit }) => {
  const [form, setForm] = useState({
    name: '',
    price: '',
    category: ''
  });

  const [errors, setErrors] = useState({});

  const validateForm = () => {
    const newErrors = {};

    if (!form.name.trim()) {
      newErrors.name = 'Name is required';
    } else if (form.name.length < 3) {
      newErrors.name = 'Name must be at least 3 characters';
    }

    if (!form.price) {
      newErrors.price = 'Price is required';
    } else if (parseFloat(form.price) <= 0) {
      newErrors.price = 'Price must be greater than 0';
    }

    if (!form.category) {
      newErrors.category = 'Category is required';
    }

    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleChange = (e) => {
    const { name, value } = e.target;
    setForm(prev => ({
      ...prev,
      [name]: value
    }));
    // Clear error for this field
    if (errors[name]) {
      setErrors(prev => ({
        ...prev,
        [name]: ''
      }));
    }
  };

  const handleSubmit = (e) => {
    e.preventDefault();

    if (validateForm()) {
      onSubmit(form);
      setForm({ name: '', price: '', category: '' });
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label>Name</label>
        <input
          name="name"
          value={form.name}
          onChange={handleChange}
          placeholder="Product name"
        />
        {errors.name && <span className="error">{errors.name}</span>}
      </div>

      <div>
        <label>Price</label>
        <input
          type="number"
          name="price"
          value={form.price}
          onChange={handleChange}
          placeholder="0.00"
        />
        {errors.price && <span className="error">{errors.price}</span>}
      </div>

      <div>
        <label>Category</label>
        <select name="category" value={form.category} onChange={handleChange}>
          <option value="">Select category</option>
          <option value="Electronics">Electronics</option>
          <option value="Clothing">Clothing</option>
        </select>
        {errors.category && <span className="error">{errors.category}</span>}
      </div>

      <button type="submit">Submit</button>
    </form>
  );
};

Q9: What is the virtual DOM and reconciliation?

Click to expand answer

Answer: Virtual DOM is an in-memory representation of the real DOM. React compares (diffs) old and new virtual DOM, then updates only changed elements.

  • State changes → new virtual DOM created
  • Diffing algorithm compares old vs new
  • Only changed elements updated in real DOM
  • Fast and efficient

Q12: How to handle errors in React?

Click to expand answer

Answer: Error Boundaries catch errors in child components. Also use try-catch in async functions.

// Error Boundary
class ErrorBoundary extends React.Component {
  constructor(props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error) {
    return { hasError: true, error };
  }

  componentDidCatch(error, errorInfo) {
    console.error('Error caught:', error, errorInfo);
  }

  render() {
    if (this.state.hasError) {
      return <h2>Something went wrong: {this.state.error.message}</h2>;
    }

    return this.props.children;
  }
}

// Use:
<ErrorBoundary>
  <ProductList />
</ErrorBoundary>

// In async/fetch:
try {
  const response = await fetch(url);
  const data = await response.json();
} catch (err) {
  setError('Failed to load data');
  console.error(err);
}

Q13: What is useReducer and when to use it?

Click to expand answer

Answer: useReducer manages complex state with multiple sub-values. Like Redux but for one component.

import { useReducer } from 'react';

const initialState = {
  products: [],
  loading: false,
  error: null
};

const reducer = (state, action) => {
  switch (action.type) {
    case 'FETCH_START':
      return { ...state, loading: true, error: null };
    case 'FETCH_SUCCESS':
      return { ...state, products: action.payload, loading: false };
    case 'FETCH_ERROR':
      return { ...state, error: action.payload, loading: false };
    case 'ADD_PRODUCT':
      return { ...state, products: [...state.products, action.payload] };
    case 'DELETE_PRODUCT':
      return {
        ...state,
        products: state.products.filter(p => p.id !== action.payload)
      };
    default:
      return state;
  }
};

const ProductList = () => {
  const [state, dispatch] = useReducer(reducer, initialState);

  const loadProducts = async () => {
    dispatch({ type: 'FETCH_START' });
    try {
      const response = await fetch('http://localhost:5000/api/products');
      const data = await response.json();
      dispatch({ type: 'FETCH_SUCCESS', payload: data });
    } catch (err) {
      dispatch({ type: 'FETCH_ERROR', payload: err.message });
    }
  };

  const handleDelete = (id) => {
    dispatch({ type: 'DELETE_PRODUCT', payload: id });
  };

  return (
    <div>
      {state.loading && <p>Loading...</p>}
      {state.error && <p>Error: {state.error}</p>}
      {state.products.map(p => (
        <div key={p.id}>
          <h3>{p.name}</h3>
          <button onClick={() => handleDelete(p.id)}>Delete</button>
        </div>
      ))}
    </div>
  );
};

Q14: How does React handle state immutability?

Click to expand answer

Answer: Never mutate state directly. Create new objects/arrays with spread operator or concat.

// ❌ WRONG - Mutating directly
state.name = 'new name';
state.products.push(newProduct);

// ✅ RIGHT - Create new objects
setState({ ...state, name: 'new name' });
setProducts([...products, newProduct]);

// Nested state
setProduct({
  ...product,
  details: {
    ...product.details,
    price: 100
  }
});

// Array operations immutably
// Add
[...products, newProduct]

// Remove
products.filter(p => p.id !== idToRemove)

// Update
products.map(p => p.id === id ? { ...p, ...updates } : p)

// Sort
[...products].sort((a, b) => a.price - b.price)

Q15: Explain React component lifecycle in functional components

Click to expand answer

Answer: Functional components use hooks to replace lifecycle methods.

const Component = () => {
  const [data, setData] = useState(null);

  // Mount (like componentDidMount)
  useEffect(() => {
    console.log('Component mounted');
    return () => console.log('Component unmounted');
  }, []);

  // Update (when dependency changes)
  useEffect(() => {
    console.log('data changed:', data);
  }, [data]);

  // Render (component body)
  return <div>{data}</div>;
};

// Lifecycle phases:
// 1. Mount: Component created
// 2. Update: State/props change
// 3. Unmount: Component removed
